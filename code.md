### 伪代码

#### 直线( y = kx + b )
    b > or < 0(左/右);  \\以下默认为右
    if (b>off) {       \\位置靠外
      if (k<10°) {       \\方向偏右
        同侧逼近;
        while (出界)          \\k<0时有几率
          r*=0.9;
      }
      else if (k<30°){   \\方向中等
        中线逼近;
        while (出界)          \\几率不大
          r*=0.9;
      }
      else{              \\方向偏左
        对侧逼近;
        while (出界)          \\几率不大？
          r*=0.9;
      }
    }
    else {             \\位置靠内
      if (k<0°) {        \\方向偏右
        同侧逼近;
        while (出界)           \\几率极小
          r*=0.9;
      }
      else if (k<25°) {  \\方向中等
        直行;
      }
      else {             \\方向偏左
        对侧逼近;
        while (出界)           \\几率不大
          r*=0.9;
      }
    }

#### 曲线( x, y, r)
    y > 0 or < 0(左/右转); \\以下默认右转
    赛道 (内 or 外) 侧;
    if (内道){
      if (k<10°){           \\方向偏右
        内切;
        while (出界){           \\几率大
          if (反转向) r*=0.9;
          else{
            if (r<L) r*=1.1;
            else r*=-1;         \\反打方向
          }
        }
      }
      else if (k<20°){      \\方向中等
        切中线;
        while (出界){            \\几率小
          if (r>L) r=L;
          else r*=1.1；
        }
      }
      else{                 \\方向偏左
        外切;
        while (出界)            \\几率小
          r*=0.9；
      }                  
    }
    else {
      if (k<0)              \\方向偏右
        内切；
      else if (k<10°){      \\方向中等
        切中线;
        while (出界)           \\几率不大
          r*=0.9;
      }
      else {                \\方向偏左
        外切;
        while (出界)           \\有几率
          r*=0.9;
      }
    }
